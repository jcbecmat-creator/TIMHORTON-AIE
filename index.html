No hay órdenes en esta sección
`; lucide.createIcons(); return; } filtrados.forEach(o => { let btnAccion = ''; if(o.estado_confirmacion === 'Borrador') { btnAccion = ``; } else { btnAccion = ``; } let colorPill = '#f1f5f9', textPill = '#475569'; if(o.estado_confirmacion === 'Borrador') { colorPill = '#e0e7ff'; textPill = '#1d4ed8'; } else if(['En proceso', 'Confirmado', 'Pendiente de Aprobación'].includes(o.estado_confirmacion)) { colorPill = '#fef3c7'; textPill = '#b45309'; } else if(o.estado_confirmacion === 'Costos Aprobados') { colorPill = '#dcfce7'; textPill = '#15803d'; } else if(o.estado_confirmacion === 'Costos Rechazados') { colorPill = '#fee2e2'; textPill = '#b91c1c'; } const tr = document.createElement('tr'); if (tabMttoActual === 'Borrador') { // Columna extra con el checkbox para borradores tr.innerHTML = ` ${o.id_ticket || 'Se asignará...'} ${o.sucursal || 'Sin asignar'} ${o.fecha_creacion || ''} ${o.estado_confirmacion} ${btnAccion} `; } else { tr.innerHTML = ` ${o.id_ticket || 'S/N'} ${o.sucursal || 'Sin asignar'} ${o.fecha_creacion || ''} ${o.estado_confirmacion} ${btnAccion} `; } tbody.appendChild(tr); }); lucide.createIcons(); } // ========================================== // 2. REEMPLAZA EL BLOQUE DE BORRADO MÚLTIPLE // ========================================== function toggleAllDrafts(checked) { document.querySelectorAll('.chk-draft').forEach(cb => cb.checked = checked); actualizarBarraBorradores(); } function actualizarBarraBorradores() { const checkboxes = document.querySelectorAll('.chk-draft:checked'); const count = checkboxes.length; document.getElementById('lbl-draft-count').innerText = `${count} folio${count !== 1 ? 's' : ''}`; // Sincronizar el checkbox general si seleccionan todos manual o desmarcan uno const allDraftsCb = document.getElementById('chk-all-drafts'); const totalCheckboxes = document.querySelectorAll('.chk-draft').length; if (allDraftsCb) { allDraftsCb.checked = (count === totalCheckboxes && totalCheckboxes > 0); } const bar = document.getElementById('delete-draft-bar'); if (count > 0) { bar.classList.add('show'); } else { bar.classList.remove('show'); } } function limpiarSeleccionBorradores() { document.querySelectorAll('.chk-draft').forEach(cb => cb.checked = false); const allDraftsCb = document.getElementById('chk-all-drafts'); if (allDraftsCb) allDraftsCb.checked = false; actualizarBarraBorradores(); } function borrarBorradoresSeleccionados() { const checkboxes = document.querySelectorAll('.chk-draft:checked'); if(checkboxes.length === 0) return; showConfirm("Borrar Órdenes", `¿Estás seguro de eliminar ${checkboxes.length} orden(es) en borrador? Esta acción no se puede deshacer.`, async () => { try { const batch = db.batch(); checkboxes.forEach(cb => { const docRef = db.collection('tickets_mantenimiento').doc(cb.value); batch.delete(docRef); }); await batch.commit(); showCustomAlert("Eliminado", "Las órdenes han sido eliminadas correctamente.", true); cargarMisOrdenes(); limpiarSeleccionBorradores(); } catch(e) { console.error("Error al borrar", e); showCustomAlert("Error", "Hubo un problema al borrar las órdenes.", false); } }); }
